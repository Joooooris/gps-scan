document.addEventListener("DOMContentLoaded", function () { const WORKER_URL = "https://calm-unit-e415.j-simon.workers.dev"; const params = new URLSearchParams(window.location.search); const qr = params.get("qr"); const id_personnel = params.get("id_personnel"); const statusElement = document.getElementById("status"); const qrElement = document.getElementById("qrText"); const employeeElement = document.getElementById("employeeText"); if (!qr || !id_personnel) { qrElement.innerText = "QR code ou employé manquant."; employeeElement.innerText = ""; statusElement.innerText = "Erreur : le lien doit contenir qr et id_personnel."; statusElement.className = "error"; } else { qrElement.innerText = "QR code : " + qr; employeeElement.innerText = "Employé : " + id_personnel; getLocationAndSend(); } function getLocationAndSend() { if (!navigator.geolocation) { statusElement.innerText = "La géolocalisation n'est pas supportée par ce téléphone."; statusElement.className = "error"; return; } navigator.geolocation.getCurrentPosition( async function (position) { const now = new Date(); const dateLocale = now.getFullYear() + "-" + String(now.getMonth() + 1).padStart(2, "0") + "-" + String(now.getDate()).padStart(2, "0"); const heureLocale = String(now.getHours()).padStart(2, "0") + ":" + String(now.getMinutes()).padStart(2, "0") + ":" + String(now.getSeconds()).padStart(2, "0"); const payload = { qr_code: Number(qr), id_personnel: Number(id_personnel), latitude: position.coords.latitude, longitude: position.coords.longitude, dates: dateLocale, heures: heureLocale, timestamp: now.toISOString() }; try { statusElement.innerText = "Envoi des données..."; const response = await fetch(WORKER_URL, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify(payload) }); const result = await response.text(); if (response.ok) { statusElement.innerText = "Heure enregistrée avec succès. Vous pouvez fermer cette page."; statusElement.className = "ok"; } else { statusElement.innerText = "Erreur lors de l'envoi : " + result; statusElement.className = "error"; } } catch (error) { statusElement.innerText = "Erreur de connexion : " + error.message; statusElement.className = "error"; } }, function (error) { statusElement.innerText = "Géolocalisation refusée ou impossible."; statusElement.className = "error"; }, { enableHighAccuracy: true, timeout: 10000, maximumAge: 0 } ); } }); // fin DOMContentLoaded
